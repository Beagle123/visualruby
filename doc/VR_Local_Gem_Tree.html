def refresh()
  self.model.clear
  result = `gem list`
  result.each_line do |line|
    gem, versions = get_name_ver(line)
    versions.split(",").each do |v|
      add_row( :pix => RUBY_ICON, :gem => "#{gem} (#{v})" )
    end
  end
end

def selected_gem
  row = self.selection.selected[1].split(" (")
  ver = row[1].gsub(")","")
  return row[0], ver
end
